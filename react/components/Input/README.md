Search input

```js
initialState = { value1: '', value2: '', value3: '' }
;<div className="w-40">
  <div className="mb5">
    <Input
      type="search"
      placeholder="Search..."
      value={state.value1}
      onChange={e => setState({ value1: e.target.value })}
      onSubmit={(e, value) => {
        e.preventDefault()
        console.log("input's:", value)
      }}
    />
  </div>

  <div className="mb5">
    <Input
      type="search"
      placeholder="Search..."
      value={state.value2}
      onChange={e => setState({ value2: e.target.value })}
      onSubmit={(e, value) => {
        e.preventDefault()
        console.log("input's value:", value)
      }}
      size="large"
    />
  </div>

  <div className="mb5">
```
